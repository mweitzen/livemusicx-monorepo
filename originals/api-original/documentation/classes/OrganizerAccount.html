import {
  Column,
  Entity,
  OneToOne,
  Relation,
  TableInheritance,
  ChildEntity,
  ManyToOne,
  ManyToMany,
  JoinTable,
  // ManyToOne,
} from 'typeorm';
import { ApiProperty } from '@nestjs/swagger';
import { VersionedResource } from '../../common/resource.entity';
import { AccountType } from '../../../enums/account-type.enum';
import { Band } from '../band.entity';
import { Venue } from '../venue.entity';
import { Musician } from '../musician.entity';
import { Organizer } from '../organizer.entity';
import { Genre } from '../../tags/genre.entity';
import { PerformerUser } from '../../user';

@Entity()
@TableInheritance({ column: 'type' })
export class Account extends VersionedResource {
  /**
   * DISCRIMINATOR
   */
  @Column({
    type: 'enum',
    enum: AccountType,
  })
  type: AccountType;

  /**
   * METADATA
   */
  @Column({ default: true })
  @ApiProperty({ example: true })
  isActive: boolean;

  @Column()
  deactivatedAt: Date;

  /**
   * BASIC INFO
   */
  @Column()
  @ApiProperty({
    description: 'Account name.',
    example: 'Account',
  })
  name: string;

  @Column()
  @ApiProperty({
    description: 'About account.',
    example: 'About',
  })
  about?: string;

  @Column()
  @ApiProperty({
    description: 'Account avatar url.',
    example: 'https://example.com/image.png',
  })
  avatarUrl?: string;

  @Column()
  @ApiProperty({
    description: 'Account thumbnail image url.',
    example: 'https://example.com/image.png',
  })
  imageUrl?: string;

  @ManyToMany(() => Genre, (genre) => genre.accounts)
  @JoinTable({ name: 'account_genres' })
  @ApiProperty()
  genres?: Relation<Genre>[];

  /**
   * SOCIAL ACCOUNTS
   */
  @Column()
  @ApiProperty()
  websiteUrl?: string;

  @Column()
  @ApiProperty()
  youtubeUrl?: string;

  @Column()
  @ApiProperty()
  facebookUrl?: string;

  @Column()
  @ApiProperty()
  twitterUrl?: string;

  @Column()
  @ApiProperty()
  instagramUrl?: string;

  /**
   * CONTACT
   */
  @Column()
  @ApiProperty()
  canEmail: boolean;

  @Column()
  @ApiProperty()
  canText: boolean;

  @Column()
  @ApiProperty()
  canCall: boolean;

  /**
   * AFFILIATES
   */
  // @ManyToMany(() => Musician)
  // affiliatedMusicians: Relation<Musician[]>;

  // @ManyToMany(() => Band)
  // affiliatedBands: Relation<Band[]>;

  // @ManyToMany(() => Organizer)
  // affiliatedOrganizers: Relation<Organizer[]>;

  // @ManyToMany(() => Venue)
  // affiliatedVenues: Relation<Venue[]>;
}

@ChildEntity(AccountType.MUSICIAN)
export class MusicianAccount extends Account {
  @ManyToOne(() => PerformerUser, (user) => user.managedAccounts)
  managedBy: Relation<PerformerUser>;

  @OneToOne(() => Musician, (musician) => musician.account)
  musician: Relation<Musician>;
}

@ChildEntity(AccountType.BAND)
export class BandAccount extends Account {
  @ManyToOne(() => PerformerUser, (user) => user.managedAccounts)
  managedBy: Relation<PerformerUser>;

  @OneToOne(() => Band, (band) => band.account)
  band: Relation<Band>;
}

@ChildEntity(AccountType.VENUE)
export class VenueAccount extends Account {
  @ManyToOne(() => PerformerUser, (user) => user.managedAccounts)
  managedBy: Relation<PerformerUser>;

  @OneToOne(() => Venue, (venue) => venue.account)
  venue: Relation<Venue>;
}

@ChildEntity(AccountType.ORGANIZER)
export class OrganizerAccount extends Account {
  @ManyToOne(() => PerformerUser, (user) => user.managedAccounts)
  managedBy: Relation<PerformerUser>;

  @OneToOne(() => Organizer, (organizer) => organizer.account)
  organizer: Relation<Organizer>;
}
