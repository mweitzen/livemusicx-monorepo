import { Account, Organizer, Venue } from './account';
import { VersionedResource } from './common';
import { User, UserRole } from './user';
import { Event } from './event';
import { PartialType } from '@nestjs/mapped-types';

/**
 * POST
 *
 *
 *
 */
export enum PostType {
  STATEMENT,
  OPINION,
  REVIEW,
  NEWSLETTER,
  ANNOUNCEMENT,
  BROADCAST,
  BULLETIN,
}
export class Post extends VersionedResource {
  title: string;
  author: User;
  content: any;
  likes: number;
  shares: number;
  comments: Comment[];
  type: PostType;
}

/**
 * COMMENT
 *
 *
 *
 */
export class Comment extends VersionedResource {
  author: User;
  content: string;
  likes: number;
  post: Post;
}

/**
 * POST - STATEMENT
 *
 *
 *
 */
export class Statement extends VersionedResource {
  author: User;
  content: string;
  likes: number;
}

/**
 * POST - OPINION
 *
 *
 *
 */
export class Opinion extends Post {
  agrees: number;
  disagrees: number;
}

/**
 * POST - REVIEW
 *
 *
 *
 */
export enum ReviewType {
  PERFORMER,
  VENUE,
  ORGANIZER,
  EVENT,
}
export class Review extends Opinion {
  reviewType: ReviewType;
}
export class PerformerReview extends Review {
  reviewType: ReviewType = ReviewType.PERFORMER;
  performer: Account;
}
export class VenueReview extends Review {
  reviewType: ReviewType = ReviewType.VENUE;
  venue: Venue;
}
export class OrganizerReview extends Review {
  reviewType: ReviewType = ReviewType.ORGANIZER;
  organizer: Organizer;
}
export class EventReview extends Review {
  reviewType: ReviewType = ReviewType.EVENT;
  event: Event;
}

/**
 * POST - NEWSLETTER
 *
 *
 *
 */
export class Newsletter extends Post {
  account: Account;
  accountType: Exclude<UserRole, UserRole.PUBLIC | UserRole.ASSOCIATE>;
}

/**
 * POST - ANNOUNCEMENT
 *
 *
 *
 */
export enum AnnouncementType {
  PRESS_RELEASE,
  MEDIA_RELEASE,
  EVENT_UPDATE,
  GENERAL,
}
export class Announcement extends Post {
  account: Account;
  accountType: Exclude<UserRole, UserRole.PUBLIC | UserRole.ASSOCIATE>;
  announcementType: AnnouncementType;
}

/**
 * POST - BULLETIN

 *
 *
 */
export enum BroadcastType {
  VENUE_SEEKING,
  ORGANIZER_SEEKING,
  PERFORMER_SEEKING,
}
export enum BroadcastAudience {
  VENUE,
  ORGANIZER,
  PERFORMER,
}
/**
 * POST - NETWORKING BROADCAST
 *
 *
 *
 */
export class Broadcast extends Post {
  broadcastType: BroadcastType;
  audience: BroadcastAudience;
  producer: Account;
  // Organizer seeking -- venue, performer
  // Venue seeking -- organizer, performer
  // Performer seeking -- organizer, venue, performer
}

export class Bulletin extends Broadcast {
  // more specific. Attached event, etc
}

/**
 *
 *
 *
 *
 *
 *
 */
export abstract class Group extends VersionedResource {
  name: string;
  description: string;
  members: User[];
  lastInteractionAt: Date;
  isActive: boolean;

  // Content
  statements: Statement[];
  opinions: Opinion[];
  broadcasts: Broadcast[];
  bulletins: Bulletin[];
}

// Networks are publicly accessible segmentations based on location.
// auto-controlled, no new networks added unless inside internal application
export class Network extends Group {
  location: any; // reference to a state,region,city,neighborhood
  parent: Network;
  children: Network[];
  threads: Thread[];
  groups: PrivateGroup[];
  // Content - made my public users, accessible to all, Only exists in a network
  reviews: Review[];
}
export class Thread {
  name: string;
  topic: string;
  network: Network;
}
export class PrivateGroup extends Group {
  isPublic: boolean;
  managedBy: User;
  administrators: User[];
  moderators: User[];
}

export class NetworkBroadcast extends Broadcast {
  network: Network;
}

export class GroupBroadcast extends Broadcast {
  group: Group;
}
