import { PartialType, OmitType, PickType } from '@nestjs/mapped-types';
import { Band, Musician, Organizer, Stage, Venue } from './account';
import { EventKeyword, Genre, VersionedResource } from './common';
import { User } from './user';

export enum EventType {
  AMBIENT,
  OPEN_MIC,
  JAM_SESSION,
  MEETUP,
}

export enum EventStatus {
  CANCELLED,
  POSTPONED,
  SCHEDULED,
  RESCHEDULED,
  UNSCHEDULED,
}

/**
 * EVENT
 *
 *
 *
 */
export class Event extends VersionedResource {
  /* METADATA */
  createdBy: User;
  lastUpdatedBy: User;

  /* PRIVILEGES */
  admins: User[];
  editors: User[];
  viewers: User[];

  /* BASIC INFO */
  title: string;
  description?: string;
  imageUrl?: string;

  /* TAGS */
  genres: Genre[];
  keywords: EventKeyword[];

  /* DATE TIME INFO */
  timeDoor?: Date;
  timeStart: Date;
  timeEnd?: Date;

  /* EVENT DETAILS */
  isPrivate: boolean;
  isFree: boolean;
  isChildFriendly: boolean;
  isHoliday: boolean;
  ageRestriction: boolean;
  minimumAge?: number;
  servesAlcohol: boolean;
  servesFood: boolean;
  requiresRegistration: boolean;
  registrationLink?: string;
  requiresTicket: boolean;

  /* EVENT LOCATION */
  venue: Venue;
  stage: Stage;

  /* EVENT PARTICIPANTS */
  musicians: Musician[];
  bands: Band[];
  organizer?: Organizer;
}

/**
 * PUBLISHED EVENT
 *
 *
 *
 */
export class PublishedEvent extends Event {
  /* PUBLISHED SPECIFIC */
  status: EventStatus;
  timeStartPrevious?: Date;
  ticketLinks?: string[];

  /* EVENT AUDIENCE */
  bookmarkedBy: User[];
  registeredUsers: User[];
}

/**
 * EVENT DRAFT
 *
 *
 *
 */
export class EventDraft extends PartialType(
  OmitType(PublishedEvent, [
    'bookmarkedBy',
    'timeStartPrevious',
    'registeredUsers',
    'status',
  ]),
) {}

/**
 * EVENT TEMPLATE
 *
 *
 *
 */
export class EventTemplate extends PartialType(Event) {
  templateName: string;
  templateDescription?: string;
  duration?: number;
}

/**
 * PAST EVENT
 *
 *
 *
 */
export class PastEvent extends PickType(PublishedEvent, [
  'admins',
  'bands',
  'bookmarkedBy',
  'createdAt',
  'createdBy',
  'description',
  'genres',
  'id',
  'imageUrl',
  'keywords',
  'musicians',
  'organizer',
  'stage',
  'timeStart',
  'title',
  'venue',
]) {}
