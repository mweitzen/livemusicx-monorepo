import {
  Genre,
  Instrument,
  PhysicalSpace,
  VenueKeyword,
  VersionedResource,
} from './common';

/**
 * ACCOUNT
 *
 *
 *
 */
export class Account extends VersionedResource {
  /* METADATA */
  isActive: boolean;
  deactivatedAt: Date;

  /* BASIC INFO */
  name: string;
  about?: string;
  avatarUrl?: string;
  imageUrl?: string;
  genres?: Genre[];

  /* SOCIAL ACCOUNTS */
  websiteUrl?: string;
  youtubeUrl?: string;
  facebookUrl?: string;
  twitterUrl?: string;
  instagramUrl?: string;

  /* CONTACT */
  canEmail: boolean;
  canText: boolean;
  canCall: boolean;

  /* AFFILIATES */
  affiliatedMusicians: Musician[];
  affiliatedBands: Band[];
  affiliatedOrganizers: Organizer[];
  affiliatedVenues: Venue[];
}

/**
 * EXTENDED ACCOUNT
 *
 *
 *
 */
export class PerformerAccount extends Account {
  spotifyUrl?: string;
  bandcampUrl?: string;
  pressKitUrl?: string;
}

export class BusinessAccount extends Account {
  googleBusinessUrl?: string;
  yelpUrl?: string;
}

/**
 * MUSICIAN
 *
 *
 *
 */
export class Musician extends PerformerAccount {
  instruments: Instrument[];
  bands: Band[];
}

/**
 * BAND
 *
 *
 *
 */
export class Band extends PerformerAccount {
  members: Musician[];
}

/**
 * VENUE
 *
 *
 *
 */
export enum VenueType {
  VENUE,
  RESTAURANT,
  CAFE,
  BAR,
  BREWERY,
  WINERY,
  DISTILLERY,
  CLUB,
  THEATER,
  GALLERY,
  AREA,
  CASINO,
  LOCATION,
  COMPLEX,
  STADIUM,
}
export enum StageType {
  STAGE,
  AREA,
  ROOM,
}
export class Venue extends BusinessAccount implements PhysicalSpace {
  capacity: number;
  /* BASIC INFO */
  type: VenueType;
  keywords: VenueKeyword[];

  /* VENUE DETAILS */
  servesAlcohol: boolean;
  servesFood: boolean;
  ageRestriction: boolean;
  minimumAge?: number;
  requiresReservation: boolean;
  reservationLink?: string;
  privateAccess: boolean;

  /* STAGES */
  stages: Stage[];

  /* LOCATION */

  /* AFFILIATES */
  affiliatedMusicians: Musician[];
  affiliatedBands: Band[];
  affiliatedOrganizers: Organizer[];
  affiliatedVenues: Venue[];
}
export class Stage extends VersionedResource implements PhysicalSpace {
  capacity: number;
  name: string;
  type: 'STAGE' | 'ROOM' | 'AREA';
  imageUrl: string;
  description: string;
  venue: Venue;
}

/**
 * ORGANIZER
 *
 *
 *
 */
export class Organizer extends BusinessAccount {
  experienceSummary?: string;
  services: string[];
}
