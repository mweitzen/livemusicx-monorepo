import { ApiProperty } from '@nestjs/swagger';
import {
  Entity,
  Column,
  PrimaryGeneratedColumn,
  CreateDateColumn,
  UpdateDateColumn,
  ManyToMany,
  TableInheritance,
} from 'typeorm';
import { PublishedEvent } from '../event/published.entity';
import { EventDraft } from '../event/draft.entity';
import { EventTemplate } from '../event/template.entity';
import { exampleUser } from '@app/data/examples/user.example';

export enum UserRole {
  PUBLIC = 'PUBLIC',
  PERFORMER = 'PERFORMER',
  VENUE = 'VENUE',
  ORGANIZER = 'ORGANIZER',
  ASSOCIATE = 'ASSOCIATE',
}

@Entity()
@TableInheritance({
  column: {
    name: 'role',
    type: 'enum',
    enum: UserRole,
    default: UserRole.PUBLIC,
  },
})
export class BaseUser {
  /**
   * METADATA
   */
  @PrimaryGeneratedColumn('uuid')
  @ApiProperty({ example: exampleUser.id })
  id: string;

  @CreateDateColumn()
  @ApiProperty({ example: exampleUser.createdAt })
  createdAt: Date;

  @UpdateDateColumn()
  @ApiProperty({ example: exampleUser.lastUpdatedAt })
  lastUpdatedAt: Date;

  @Column({ default: true })
  @ApiProperty({ example: true })
  isActive: boolean;

  /**
   * CREDENTIALS
   */
  // @Column({
  //   type: 'enum',
  //   enum: UserRole,
  //   default: UserRole.PUBLIC,
  // })
  // role: UserRole;

  @Column({ unique: true })
  @ApiProperty({ example: exampleUser.username })
  username: string;

  @Column({ unique: true, nullable: true })
  @ApiProperty({ example: exampleUser.email })
  email: string;

  @Column({ select: false })
  password: string;

  /**
   * BASIC INFO
   */
  @Column()
  @ApiProperty({ example: exampleUser.firstName })
  firstName: string;

  @Column()
  @ApiProperty({ example: exampleUser.lastName })
  lastName: string;

  /**
   * EVENT AUDIENCE
   */
  @ManyToMany(() => PublishedEvent, (event) => event.bookmarkedBy)
  bookmarkedEvents: PublishedEvent[];

  @ManyToMany(() => PublishedEvent, (event) => event.registeredUsers)
  registeredEvents: PublishedEvent[];

  /**
   * EVENTUALLY MOVE TO ADMIN
   */
  // @OneToMany(() => PublishedEvent, (event) => event.lastUpdatedBy)
  eventEdits: PublishedEvent[];

  // @OneToMany(() => PublishedEvent, (event) => event.createdBy)
  createdEvents: PublishedEvent[];

  // @ManyToMany(() => PublishedEvent, (event) => event.admins)
  administratorOf: PublishedEvent[];

  // @ManyToMany(() => PublishedEvent, (event) => event.editors)
  editorOf: PublishedEvent[];

  // @ManyToMany(() => PublishedEvent, (event) => event.viewers)
  viewerOf: PublishedEvent[];

  // @OneToMany(() => EventDraft, (event) => event.lastUpdatedBy)
  eventDraftEdits: EventDraft[];

  // @OneToMany(() => EventDraft, (event) => event.createdBy)
  createdEventDrafts: EventDraft[];

  // @ManyToMany(() => EventDraft, (event) => event.admins)
  administratorOfDraft: EventDraft[];

  // @ManyToMany(() => EventDraft, (event) => event.editors)
  editorOfDraft: EventDraft[];

  // @ManyToMany(() => EventDraft, (event) => event.viewers)
  viewerOfDraft: EventDraft[];

  // @OneToMany(() => EventTemplate, (event) => event.lastUpdatedBy)
  eventTemplateEdits: EventTemplate[];

  // @OneToMany(() => EventTemplate, (event) => event.createdBy)
  createdEventTemplates: EventTemplate[];

  // @ManyToMany(() => EventTemplate, (event) => event.admins)
  administratorOfTemplate: EventTemplate[];

  // @ManyToMany(() => EventTemplate, (event) => event.editors)
  editorOfTemplate: EventTemplate[];

  // @ManyToMany(() => EventTemplate, (event) => event.viewers)
  viewerOfTemplate: EventTemplate[];
}
